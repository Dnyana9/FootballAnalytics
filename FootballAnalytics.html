Conversion of Wage and Value attributes

fifa <- fifa %>%
        mutate(Multiplier = ifelse(str_detect(Value, "K"), 1000, ifelse(str_detect(Value, "M"), 1000000, 1))) %>%
        mutate(ValueNumeric = as.numeric(str_extract(Value, "[[:digit:]]+\\.*[[:digit:]]*")) * Multiplier) %>%
        mutate(Position = ifelse(is.na(Position), "Unknown", Position))
fifa <- fifa %>%
        mutate(WMultiplier = ifelse(str_detect(Wage, "K"), 1000, ifelse(str_detect(Wage, "M"), 1000000, 1))) %>%
        mutate(WageNumeric = as.numeric(str_extract(Wage, "[[:digit:]]+\\.*[[:digit:]]*")) * WMultiplier)
